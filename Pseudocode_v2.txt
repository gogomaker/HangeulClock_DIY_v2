void loop() {
    기타 세팅...

    상태_모드전환버튼 = 버튼감지함수(모드전환버튼)
    상태_LED버튼 = 버튼감지함수(LED버튼)
    상태_시간버튼 = 버튼감지함수(시간버튼)
    상태_알람버튼 = 버튼감지함수(알람버튼)
    
    if(모드전환버튼 눌렸으면) {
        모드전환;
    }
    if(LED버튼 짧게 눌렸으면) {
        LED 밝기 전환;
    }
    if(시간모드라면) {
        if(LED버튼 길게 눌렸으면) {
            LED색상 변경
        }
        if(시간버튼 짧게 눌렸으면) {
            if(시간수정모드 활성) {
                시단위 1추가
            }
        }
        if(시간버튼 길게 눌렸으면) {
            시간수정모드 = !시간수정모드; 
            LED 점멸
        }
        if(알람버튼 짧게 눌렸으면) {
            시간수정모드 활성 ? 분단위 1추가 : 알람 활성화 여부 표시
        }
    }

    switch(모드) {
        case 시간모드:
            시간표시();
            break;
        case 온도모드:
            온도표시();
            break;
        case 습도모드:
            습도표시();
            break;
    }
}

    }
}
